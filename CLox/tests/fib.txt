
class Test {
 assert(name, a, b) {
	if (a == b) {
		print "PASSED " + name + " ";
		print a;
		print " == ";
		print b;
		print "\n";
	} else {
		print "FAILED " + name + " ";
		print a;
		print " != ";
		print b;
		print "\n";
	}
  }
}

var test = Test();


fun fib(n)
{
     var a = 0;
     var b = 1;
     var  c;
     var i;

    if( n == 0)
        return a;
    for(i = 2; i <= n; i = i + 1)
    {
    c = a + b;
    a = b;
    b = c;
    }
    return b;
}
 

// fibonachi table =  https://r-knott.surrey.ac.uk/Fibonacci/fibtable.html 
// 0 = 0, 7=13, 15=610, 72=498454011879264 80=23416728348467685

test.assert("Fib(0)", fib(0), 0);
test.assert("Fib(7)", fib(7), 13);
test.assert("Fib(15)", fib(15), 610);
test.assert("Fib(72)", fib(72), 498454011879264);
test.assert("Fib(100)", fib(80), 23416728348467685);
